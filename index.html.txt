```react
import React, { useState, useEffect } from 'react';
import { initializeApp } from 'firebase/app';
import { 
  getFirestore, 
  doc, 
  setDoc, 
  getDoc, 
  collection, 
  onSnapshot 
} from 'firebase/firestore';
import { 
  getAuth, 
  signInAnonymously, 
  signInWithCustomToken,
  onAuthStateChanged 
} from 'firebase/auth';
import { 
  Trophy, 
  Settings, 
  Lock, 
  Eye, 
  Play, 
  Plus, 
  Trash2, 
  CheckCircle, 
  AlertCircle, 
  HelpCircle, 
  Coins, 
  User, 
  LogOut, 
  Upload,
  ArrowRight,
  Sparkles,
  Compass,
  Home,
  RotateCcw,
  Download
} from 'lucide-react';

// === CONFIGURATION & INITIALIZATION ===
const firebaseConfig = JSON.parse(__firebase_config);
const app = initializeApp(firebaseConfig);
const auth = getAuth(app);
const db = getFirestore(app);
const appId = typeof __app_id !== 'undefined' ? __app_id : 'sabahan-kita-game';

export default function App() {
  // --- STATE ---
  const [user, setUser] = useState(null);
  const [isAdmin, setIsAdmin] = useState(false);
  const [activeTab, setActiveTab] = useState('start'); // 'start' | 'game' | 'admin'
  
  // Game & Admin Data dari Firestore
  const [questions, setQuestions] = useState([]);
  const [config, setConfig] = useState({
    adminPassword: 'admin', // Default password
    rewardVideoUrl: 'https://www.youtube.com/watch?v=dQw4w9WgXcQ', // Default video
    videoDuration: 10 // Simulasi saat tontonan video
  });

  // Player State (localStorage supaya progress pemain kekal)
  const [playerPoints, setPlayerPoints] = useState(() => {
    const saved = localStorage.getItem('sabahan_points');
    return saved ? parseInt(saved, 10) : 100; // Mula dengan 100 point percuma
  });
  
  const [currentQuestionIndex, setCurrentQuestionIndex] = useState(() => {
    const saved = localStorage.getItem('sabahan_current_index');
    return saved ? parseInt(saved, 10) : 0;
  });

  // Jawapan semasa pemain
  const [selectedLetters, setSelectedLetters] = useState([]); 
  const [revealedIndices, setRevealedIndices] = useState([]); 
  const [hasRequestedAnswer, setHasRequestedAnswer] = useState(false);

  // Admin Login State
  const [adminInputPassword, setAdminInputPassword] = useState('');
  const [isPasswordCorrect, setIsPasswordCorrect] = useState(false);
  const [adminErrorMessage, setAdminErrorMessage] = useState('');

  // Admin Editor State (Borang Tambah/Edit Soalan)
  const [newQuestionText, setNewQuestionText] = useState('');
  const [newAnswer, setNewAnswer] = useState('');
  const [newImageBase64, setNewImageBase64] = useState('');
  const [editingConfig, setEditingConfig] = useState({ ...config });
  const [newAdminPassword, setNewAdminPassword] = useState('');
  const [confirmAdminPassword, setConfirmAdminPassword] = useState('');
  const [adminSuccessMsg, setAdminSuccessMsg] = useState('');

  // Video Reward Modal State
  const [showVideoModal, setShowVideoModal] = useState(false);
  const [isVideoPlaying, setIsVideoPlaying] = useState(false);
  const [videoTimer, setVideoTimer] = useState(0);
  const [videoWatchedSuccessfully, setVideoWatchedSuccessfully] = useState(false);

  // General Notification Modal
  const [notification, setNotification] = useState(null);

  // PWA Install State (Pemasangan terus ke skrin telefon)
  const [deferredPrompt, setDeferredPrompt] = useState(null);
  const [showInstallBtn, setShowInstallBtn] = useState(false);

  // --- EFFECT 1: FIREBASE AUTH ---
  useEffect(() => {
    const initAuth = async () => {
      try {
        if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
          await signInWithCustomToken(auth, __initial_auth_token);
        } else {
          await signInAnonymously(auth);
        }
      } catch (error) {
        console.error("Ralat Log Masuk Auth:", error);
      }
    };
    initAuth();

    const unsubscribe = onAuthStateChanged(auth, (usr) => {
      setUser(usr);
    });
    return () => unsubscribe();
  }, []);

  // --- EFFECT 2: READ GAME DATA FROM FIRESTORE (REAL-TIME) ---
  useEffect(() => {
    if (!user) return;

    const configDocRef = doc(db, 'artifacts', appId, 'public', 'data', 'config');
    const unsubConfig = onSnapshot(configDocRef, (docSnap) => {
      if (docSnap.exists()) {
        const data = docSnap.data();
        setConfig(data);
        setEditingConfig(data);
      } else {
        setDoc(configDocRef, config);
      }
    }, (error) => {
      console.error("Ralat membaca konfigurasi:", error);
    });

    const questionsDocRef = doc(db, 'artifacts', appId, 'public', 'data', 'questions');
    const unsubQuestions = onSnapshot(questionsDocRef, (docSnap) => {
      if (docSnap.exists()) {
        const data = docSnap.data();
        if (data.list) {
          setQuestions(data.list);
        }
      } else {
        const defaultQuestions = [
          {
            id: 'q1',
            question: 'Apakah nama haiwan ikonik di Sabah yang terkenal dengan hidung panjangnya?',
            answer: 'PROBOSCIS',
            image: '' 
          },
          {
            id: 'q2',
            question: 'Gunung tertinggi di Malaysia yang terletak di Sabah ialah Gunung...',
            answer: 'KINABALU',
            image: ''
          }
        ];
        setDoc(questionsDocRef, { list: defaultQuestions });
      }
    }, (error) => {
      console.error("Ralat membaca soalan:", error);
    });

    return () => {
      unsubConfig();
      unsubQuestions();
    };
  }, [user]);

  // --- EFFECT 3: PWA INSTALLATION DETECTOR ---
  useEffect(() => {
    const handleBeforeInstallPrompt = (e) => {
      // Halang pelayar web daripada menunjukkan prompt lalai secara automatik
      e.preventDefault();
      // Simpan event supaya boleh dicetuskan kemudian melalui butang tersuai kita
      setDeferredPrompt(e);
      // Tunjukkan butang "Pasang Aplikasi" di skrin utama game
      setShowInstallBtn(true);
    };

    window.addEventListener('beforeinstallprompt', handleBeforeInstallPrompt);

    // Semak jika aplikasi sudah sedia dipasang (Standalone mode)
    if (window.matchMedia('(display-mode: standalone)').matches) {
      setShowInstallBtn(false);
    }

    return () => {
      window.removeEventListener('beforeinstallprompt', handleBeforeInstallPrompt);
    };
  }, []);

  // Simpan progress mata & indeks soalan
  useEffect(() => {
    localStorage.setItem('sabahan_points', playerPoints.toString());
  }, [playerPoints]);

  useEffect(() => {
    localStorage.setItem('sabahan_current_index', currentQuestionIndex.toString());
    setSelectedLetters([]);
    setRevealedIndices([]);
    setHasRequestedAnswer(false);
  }, [currentQuestionIndex]);

  // --- FUNCTION: HANDLE PWA INSTALL ---
  const handleInstallApp = async () => {
    if (!deferredPrompt) return;
    // Tunjukkan promt pemasangan daripada pelayar
    deferredPrompt.prompt();
    // Tunggu keputusan tindakan pemain (Setuju pasang atau Batal)
    const { outcome } = await deferredPrompt.userChoice;
    if (outcome === 'accepted') {
      triggerNotification('success', 'Aplikasi sedang dipasang ke skrin telefon anda! 🎉');
      setShowInstallBtn(false);
    } else {
      triggerNotification('error', 'Pemasangan dibatalkan.');
    }
    setDeferredPrompt(null);
  };

  // --- UTILITY: SHOW CUSTOM NOTIFICATION ---
  const triggerNotification = (type, message) => {
    setNotification({ type, message });
    setTimeout(() => {
      setNotification(null);
    }, 4000);
  };

  // --- ADMIN AUTHENTICATION ---
  const handleAdminLogin = (e) => {
    e.preventDefault();
    if (adminInputPassword === config.adminPassword) {
      setIsPasswordCorrect(true);
      setAdminErrorMessage('');
      triggerNotification('success', 'Selamat datang Admin!');
    } else {
      setAdminErrorMessage('Kata laluan salah! Sila cuba lagi.');
    }
  };

  const handleAdminLogout = () => {
    setIsPasswordCorrect(false);
    setAdminInputPassword('');
    setIsAdmin(false);
    setActiveTab('start');
  };

  // --- ADMIN: UPDATE CONFIGURATION ---
  const handleUpdateConfig = async (e) => {
    e.preventDefault();
    try {
      const configDocRef = doc(db, 'artifacts', appId, 'public', 'data', 'config');
      
      let updatedData = {
        rewardVideoUrl: editingConfig.rewardVideoUrl,
        videoDuration: editingConfig.videoDuration || 10,
        adminPassword: config.adminPassword 
      };

      if (newAdminPassword) {
        if (newAdminPassword !== confirmAdminPassword) {
          triggerNotification('error', 'Kata laluan baru tidak sepadan!');
          return;
        }
        updatedData.adminPassword = newAdminPassword;
      }

      await setDoc(configDocRef, updatedData);
      setNewAdminPassword('');
      setConfirmAdminPassword('');
      setAdminSuccessMsg('Konfigurasi dan Kata Laluan berjaya disimpan!');
      setTimeout(() => setAdminSuccessMsg(''), 4000);
      triggerNotification('success', 'Tetapan berjaya dikemaskini!');
    } catch (err) {
      console.error(err);
      triggerNotification('error', 'Gagal menyimpan tetapan.');
    }
  };

  // --- ADMIN: IMAGE UPLOAD TO BASE64 ---
  const handleImageChange = (e) => {
    const file = e.target.files[0];
    if (file) {
      if (file.size > 800000) { 
        triggerNotification('error', 'Saiz gambar terlalu besar! Sila gunakan gambar di bawah 800KB.');
        return;
      }
      const reader = new FileReader();
      reader.onloadend = () => {
        setNewImageBase64(reader.result);
      };
      reader.readAsDataURL(file);
    }
  };

  // --- ADMIN: ADD NEW QUESTION ---
  const handleAddQuestion = async (e) => {
    e.preventDefault();
    if (!newQuestionText.trim() || !newAnswer.trim()) {
      triggerNotification('error', 'Sila isi soalan dan jawapan!');
      return;
    }

    const cleanAnswer = newAnswer.toUpperCase().replace(/[^A-Z]/g, '');
    if (cleanAnswer.length === 0) {
      triggerNotification('error', 'Jawapan mestilah mengandungi huruf A-Z sahaja!');
      return;
    }

    const newQ = {
      id: 'q_' + Date.now(),
      question: newQuestionText.trim(),
      answer: cleanAnswer,
      image: newImageBase64
    };

    const updatedList = [...questions, newQ];
    
    try {
      const questionsDocRef = doc(db, 'artifacts', appId, 'public', 'data', 'questions');
      await setDoc(questionsDocRef, { list: updatedList });
      
      setNewQuestionText('');
      setNewAnswer('');
      setNewImageBase64('');
      const fileInput = document.getElementById('admin-image-input');
      if (fileInput) fileInput.value = '';

      triggerNotification('success', 'Soalan berjaya ditambah!');
    } catch (err) {
      console.error(err);
      triggerNotification('error', 'Gagal menambah soalan ke pangkalan data.');
    }
  };

  // --- ADMIN: DELETE QUESTION ---
  const handleDeleteQuestion = async (id) => {
    const updatedList = questions.filter(q => q.id !== id);
    try {
      const questionsDocRef = doc(db, 'artifacts', appId, 'public', 'data', 'questions');
      await setDoc(questionsDocRef, { list: updatedList });
      triggerNotification('success', 'Soalan berjaya dipadam!');
      
      if (currentQuestionIndex >= updatedList.length && updatedList.length > 0) {
        setCurrentQuestionIndex(updatedList.length - 1);
      }
    } catch (err) {
      console.error(err);
      triggerNotification('error', 'Gagal memadam soalan.');
    }
  };

  // --- GAME LOGIC ---
  const currentQuestion = questions[currentQuestionIndex] || null;

  const getLetterPool = () => {
    if (!currentQuestion) return [];
    const ans = currentQuestion.answer;
    
    let pool = ans.split('');
    const alphabet = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ';
    while (pool.length < 14) {
      const randomChar = alphabet[Math.floor(Math.random() * alphabet.length)];
      pool.push(randomChar);
    }
    
    const seed = currentQuestion.id.charCodeAt(currentQuestion.id.length - 1) || 1;
    let seededPool = pool.map((char, index) => ({ char, sortVal: (index * seed * 13) % 97, originalIndex: index }));
    seededPool.sort((a, b) => a.sortVal - b.sortVal);
    
    return seededPool.map(item => item.char);
  };

  const letterPool = React.useMemo(() => getLetterPool(), [currentQuestion]);

  const handleSelectLetter = (char, indexInPool) => {
    if (!currentQuestion) return;
    const maxLen = currentQuestion.answer.length;
    
    if (selectedLetters.length >= maxLen) return;
    if (selectedLetters.some(item => item.indexInPool === indexInPool)) return;

    const nextSelected = [...selectedLetters, { indexInPool, char }];
    setSelectedLetters(nextSelected);
    checkAnswerCompletion(nextSelected);
  };

  const handleRemoveLetter = (indexInSelected) => {
    const updated = [...selectedLetters];
    updated.splice(indexInSelected, 1);
    setSelectedLetters(updated);
  };

  const checkAnswerCompletion = (currentSelected) => {
    if (!currentQuestion) return;
    const ans = currentQuestion.answer;
    if (currentSelected.length === ans.length) {
      const playersAns = currentSelected.map(item => item.char).join('');
      if (playersAns === ans) {
        setPlayerPoints(prev => prev + 20);
        triggerNotification('success', 'Tahniah! Jawapan anda BETUL! (+20 Point)');
        
        setTimeout(() => {
          if (currentQuestionIndex < questions.length - 1) {
            setCurrentQuestionIndex(prev => prev + 1);
          } else {
            triggerNotification('success', 'Hebat! Anda telah menyelesaikan SEMUA soalan "Sabahan Kita"!');
          }
        }, 1500);
      } else {
        triggerNotification('error', 'Jawapan kurang tepat. Sila cuba lagi!');
      }
    }
  };

  // --- HINT 1: REVEAL ONE LETTER ---
  const handleRequestHint = () => {
    if (!currentQuestion) return;
    if (playerPoints < 10) {
      triggerNotification('error', 'Mata tidak mencukupi untuk meminta pembayang! Sila tonton video untuk tambah point.');
      return;
    }

    const ans = currentQuestion.answer;
    let unrevealed = [];
    for (let i = 0; i < ans.length; i++) {
      if (!revealedIndices.includes(i)) {
        unrevealed.push(i);
      }
    }

    if (unrevealed.length === 0) {
      triggerNotification('error', 'Semua huruf telah dibongkar!');
      return;
    }

    const randomIndex = unrevealed[Math.floor(Math.random() * unrevealed.length)];
    setRevealedIndices(prev => [...prev, randomIndex]);
    setPlayerPoints(prev => prev - 10);
    triggerNotification('success', `Huruf ke-${randomIndex + 1} ialah "${ans[randomIndex]}"! (-10 Point)`);
  };

  // --- HINT 2: REVEAL FULL ANSWER ---
  const handleRequestFullAnswer = () => {
    if (!currentQuestion) return;
    if (playerPoints < 50) {
      triggerNotification('error', 'Mata tidak mencukupi untuk mendapatkan jawapan penuh! Sila tonton video.');
      return;
    }

    setPlayerPoints(prev => prev - 50);
    setHasRequestedAnswer(true);
    
    const ans = currentQuestion.answer;
    const fullSelected = ans.split('').map((char, idx) => ({
      indexInPool: -1 - idx,
      char: char
    }));
    setSelectedLetters(fullSelected);

    triggerNotification('success', `Jawapan penuh dibongkar: "${ans}"! (-50 Point)`);
  };

  const handleNextQuestionAfterReveal = () => {
    if (currentQuestionIndex < questions.length - 1) {
      setCurrentQuestionIndex(prev => prev + 1);
    } else {
      triggerNotification('success', 'Selesai! Anda telah menamatkan semua soalan.');
    }
  };

  // --- REWARD VIDEO AD SYSTEM ---
  const startWatchingVideo = () => {
    setShowVideoModal(true);
    setIsVideoPlaying(true);
    setVideoTimer(config.videoDuration || 10);
    setVideoWatchedSuccessfully(false);
  };

  useEffect(() => {
    let interval = null;
    if (isVideoPlaying && videoTimer > 0) {
      interval = setInterval(() => {
        setVideoTimer(prev => prev - 1);
      }, 1000);
    } else if (videoTimer === 0 && isVideoPlaying) {
      setIsVideoPlaying(false);
      setVideoWatchedSuccessfully(true);
    }
    return () => clearInterval(interval);
  }, [isVideoPlaying, videoTimer]);

  const handleClaimReward = () => {
    if (videoWatchedSuccessfully) {
      setPlayerPoints(prev => prev + 25);
      setShowVideoModal(false);
      setVideoWatchedSuccessfully(false);
      triggerNotification('success', 'Terima kasih menonton! Anda mendapat +25 Point ganjaran! ??');
    }
  };

  const handleCancelVideo = () => {
    setIsVideoPlaying(false);
    setShowVideoModal(false);
    triggerNotification('error', 'Tontonan dibatalkan! Tiada point diberikan kerana anda melangkau video.');
  };

  const handleResetGameProgress = () => {
    setCurrentQuestionIndex(0);
    setPlayerPoints(100);
    setSelectedLetters([]);
    setRevealedIndices([]);
    setHasRequestedAnswer(false);
    triggerNotification('success', 'Progress permainan telah di-reset semula!');
  };

  return (
    <div className="min-h-screen bg-slate-900 text-white font-sans selection:bg-yellow-400 selection:text-slate-900 flex flex-col justify-between">
      
      {/* HEADER UTAMA */}
      <header className="sticky top-0 z-40 backdrop-blur-md bg-slate-900/80 border-b border-slate-800 px-4 py-3 shadow-lg">
        <div className="max-w-4xl mx-auto flex items-center justify-between">
          <button 
            onClick={() => setActiveTab('start')}
            className="flex items-center gap-2 text-left hover:opacity-90 transition"
          >
            <div className="bg-yellow-500 text-slate-950 p-2 rounded-xl font-black text-xl tracking-wider shadow-inner transform rotate-[-2deg]">
              SK
            </div>
            <div>
              <h1 className="font-extrabold text-xl tracking-tight text-yellow-400">Sabahan Kita</h1>
              <p className="text-xs text-slate-400 font-medium">Budaya, Dialek & Teka-Teki</p>
            </div>
          </button>

          <div className="flex items-center gap-2">
            {/* Skor Point */}
            <div className="bg-slate-800 border border-slate-700 rounded-full px-3 py-1 flex items-center gap-2 shadow-sm">
              <Coins className="w-4 h-4 text-yellow-400 fill-yellow-400" />
              <span className="font-bold text-sm text-yellow-300">{playerPoints} <span className="text-xs text-slate-400 font-normal">pts</span></span>
            </div>

            {/* Menu Admin */}
            <button
              onClick={() => {
                if (activeTab === 'admin') {
                  setActiveTab('start');
                } else {
                  setActiveTab('admin');
                }
              }}
              className={`p-2 rounded-xl transition-all duration-200 ${
                activeTab === 'admin' 
                  ? 'bg-yellow-500 text-slate-900 shadow-lg scale-105' 
                  : 'bg-slate-800 text-slate-300 hover:bg-slate-700'
              }`}
              title="Tetapan Admin"
            >
              <Settings className="w-5 h-5" />
            </button>
          </div>
        </div>
      </header>

      {/* NOTIFICATION TOAST */}
      {notification && (
        <div className="fixed top-20 right-4 left-4 md:left-auto md:w-96 z-50 transition-all duration-300 transform translate-y-0">
          <div className={`p-4 rounded-xl border flex items-start gap-3 shadow-2xl ${
            notification.type === 'success' 
              ? 'bg-emerald-950/90 border-emerald-500 text-emerald-200' 
              : 'bg-rose-950/90 border-rose-500 text-rose-200'
          }`}>
            {notification.type === 'success' ? (
              <CheckCircle className="w-5 h-5 text-emerald-400 shrink-0 mt-0.5" />
            ) : (
              <AlertCircle className="w-5 h-5 text-rose-400 shrink-0 mt-0.5" />
            )}
            <div>
              <p className="font-bold text-sm">{notification.type === 'success' ? 'Berjaya!' : 'Perhatian!'}</p>
              <p className="text-xs opacity-90">{notification.message}</p>
            </div>
          </div>
        </div>
      )}

      {/* VIEW UTAMA */}
      <main className="max-w-4xl mx-auto px-4 py-6 flex-grow w-full flex flex-col justify-center">
        
        {/* ==================== TAB A: SCREEN PERMULAAN (START SCREEN) ==================== */}
        {activeTab === 'start' && (
          <div className="text-center space-y-6 py-4 max-w-lg mx-auto w-full">
            
            {/* LOGO */}
            <div className="relative inline-block">
              <div className="absolute inset-0 bg-yellow-500/20 blur-3xl rounded-full" />
              <div className="relative bg-gradient-to-tr from-slate-800 to-slate-950 p-6 rounded-full border-2 border-slate-800 shadow-2xl">
                <Compass className="w-20 h-20 text-yellow-400 stroke-[1.5]" />
              </div>
            </div>

            {/* INTRO TEXT */}
            <div className="space-y-2">
              <h2 className="text-3xl font-black tracking-tight text-white">
                Bah, <span className="text-yellow-400">Sedia Sudah?</span>
              </h2>
              <p className="text-slate-400 text-xs leading-relaxed px-4">
                Uji kehebatan minda anda mengenai dialek Sabah, tempat menarik, serta budaya tradisi "Negeri Di Bawah Bayu".
              </p>
            </div>

            {/* PWA INSTALL BANNER (Hanya muncul jika dibuka di browser Google Chrome/Safari dan belum di-install) */}
            {showInstallBtn && (
              <div className="bg-gradient-to-r from-blue-950 to-indigo-950 border border-blue-800 p-4 rounded-2xl text-left flex items-center justify-between gap-3 animate-pulse shadow-xl">
                <div className="space-y-1">
                  <p className="text-xs font-bold text-blue-300 flex items-center gap-1">
                    <Sparkles className="w-3.5 h-3.5 text-yellow-400" /> Pasang Pada Telefon
                  </p>
                  <p className="text-[10px] text-slate-300 leading-normal">
                    Pasang game ini terus ke skrin telefon anda untuk bermain dengan lebih lancar!
                  </p>
                </div>
                <button
                  onClick={handleInstallApp}
                  className="bg-blue-500 hover:bg-blue-400 text-white font-extrabold px-3.5 py-2 rounded-xl text-xs flex items-center gap-1 shadow-md shrink-0 active:scale-95 transition"
                >
                  <Download className="w-3.5 h-3.5" /> PASANG
                </button>
              </div>
            )}

            {/* SEKSYEN MAKLUMAT LEVEL SEMASA */}
            <div className="bg-slate-800/40 border border-slate-800 p-4 rounded-2xl flex items-center justify-between text-left">
              <div>
                <p className="text-xs text-slate-500 font-semibold uppercase">Progress Semasa</p>
                <p className="text-sm font-bold text-slate-100">
                  {questions.length > 0 ? `Soalan ${currentQuestionIndex + 1} daripada ${questions.length}` : 'Tiada soalan sedia ada'}
                </p>
              </div>
              <div className="text-right">
                <p className="text-xs text-slate-500 font-semibold uppercase">Simpanan Point</p>
                <p className="text-sm font-black text-yellow-400 flex items-center gap-1 justify-end">
                  <Coins className="w-3.5 h-3.5 fill-yellow-400 text-yellow-400" /> {playerPoints} Pts
                </p>
              </div>
            </div>

            {/* BUTANG UTAMA */}
            <div className="space-y-2.5">
              <button
                onClick={() => {
                  if (questions.length === 0) {
                    triggerNotification('error', 'Tiada soalan sedia ada. Sila masuk ke Panel Admin untuk menambah soalan.');
                  } else {
                    setActiveTab('game');
                  }
                }}
                className="w-full bg-gradient-to-r from-yellow-500 to-amber-500 hover:from-yellow-400 hover:to-amber-400 text-slate-950 font-black py-3.5 px-6 rounded-2xl flex items-center justify-center gap-2 shadow-xl hover:scale-[1.01] transition duration-200 text-sm"
              >
                <Play className="w-4 h-4 fill-slate-950 text-slate-950" />
                MULA BERMAIN SEKARANG
              </button>

              <button
                onClick={startWatchingVideo}
                className="w-full bg-slate-800 hover:bg-slate-700 text-slate-200 font-extrabold py-3 px-6 rounded-2xl flex items-center justify-center gap-2 border border-slate-700 transition text-xs"
              >
                <Coins className="w-4 h-4 text-yellow-400 fill-yellow-400" />
                Tonton Video Untuk Tambah Point (+25 Pts)
              </button>
            </div>

            {/* RESET PROGRESS */}
            {currentQuestionIndex > 0 && (

              <a href={config.rewardVideoUrl} target="_blank" rel="noopener noreferrer" className="hover:underline flex items-center justify-center gap-1.5">
                Buka Link Video <Sparkles className="w-3.5 h-3.5 text-yellow-400" />
              </a>
            </div>

            {isVideoPlaying ? (
              <div className="space-y-2 py-2">
                <div className="text-3xl font-black text-amber-400">{videoTimer}s</div>
                <p className="text-[10px] text-slate-500">Sila kekalkan tontonan video ini di tetingkap sebelah...</p>
                <div className="w-full bg-slate-950 h-1.5 rounded-full overflow-hidden">
                  <div 
                    className="bg-amber-500 h-full transition-all duration-1000" 
                    style={{ width: `${((config.videoDuration - videoTimer) / config.videoDuration) * 100}%` }}
                  />
                </div>
              </div>
            ) : (
              <div className="bg-emerald-950/50 border border-emerald-900 p-3 rounded-2xl">
                <p className="text-xs font-bold text-emerald-400 flex items-center justify-center gap-1">
                  <CheckCircle className="w-4 h-4" /> Tontonan Selesai!
                </p>
                <p className="text-[10px] text-slate-300 mt-1">Sedia untuk menuntut 25 point ganjaran.</p>
              </div>
            )}

            <div className="flex gap-2 pt-2">
              <button
                onClick={handleCancelVideo}
                className="flex-1 bg-slate-800 hover:bg-slate-700 text-slate-400 font-bold py-2.5 rounded-xl text-xs transition"
              >
                Batal (Tiada Point)
              </button>

              <button
                onClick={handleClaimReward}
                disabled={isVideoPlaying}
                className={`flex-1 font-black py-2.5 rounded-xl text-xs transition ${
                  isVideoPlaying 
                    ? 'bg-slate-800 text-slate-600 cursor-not-allowed' 
                    : 'bg-emerald-500 hover:bg-emerald-400 text-slate-950'
                }`}
              >
                Tuntut 25 Point
              </button>
            </div>

          </div>
        </div>
      )}

    </div>
  );
}
```
